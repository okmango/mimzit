1. **Mimzit/Features/Transcription/TranscribeButtonView.swift** — Per UI-SPEC Component Inventory "TranscribeButton". Three-state view:

   ```swift
   import SwiftUI

   enum TranscribeState {
       case idle
       case inProgress
       case complete
       case error(String)
   }

   struct TranscribeButtonView: View {
       let state: TranscribeState
       let onTranscribe: () -> Void

       var body: some View {
           switch state {
           case .idle:
               Button(action: onTranscribe) {
                   Label("Transcribe Audio", systemImage: "waveform.badge.sparkles")
               }
               .buttonStyle(.bordered)
               .tint(Theme.accent)

           case .inProgress:
               HStack(spacing: 8) {
                   ProgressView()
                       .tint(Theme.accent)
                   Text("Transcribing...")
                       .foregroundStyle(.secondary)
               }

           case .complete:
               Label("Transcript ready", systemImage: "checkmark.circle.fill")
                   .foregroundStyle(Theme.transcriptReady)
                   .font(.caption)

           case .error(let message):
               VStack(alignment: .leading, spacing: 8) {
                   Button(action: onTranscribe) {
                       Label("Transcribe Audio", systemImage: "waveform.badge.sparkles")
                   }
                   .buttonStyle(.bordered)
                   .tint(Theme.accent)

                   Text(message)
                       .font(.caption)
                       .foregroundStyle(.red)
               }
           }
       }
   }
   ```

   State mapping per UI-SPEC:
   - **idle**: `Button("Transcribe Audio")` with `waveform.badge.sparkles`, `.bordered`, accent tint
   - **inProgress**: `ProgressView` (small, accent) + "Transcribing..." — NOT tappable
   - **complete**: `Label("Transcript ready", systemImage: "checkmark.circle.fill")` in `.systemGreen`, `.caption`
   - **error**: Same button as idle + inline error message below in red `.caption`

2. **Mimzit/Features/Transcription/APIKeyPromptSheet.swift** — Per UI-SPEC Screen 6 and D-13 (lazy trigger):

   ```swift
   import SwiftUI

   struct APIKeyPromptSheet: View {
       let onSave: () -> Void
       @Environment(\.dismiss) private var dismiss
       @State private var apiKey = ""
       @State private var showAPIKey = false

       var body: some View {
           NavigationStack {
               Form {
                   Section {
                       HStack {
                           if showAPIKey {
                               TextField("sk-...", text: $apiKey)
                                   .textContentType(.password)
                                   .autocorrectionDisabled()
                                   .textInputAutocapitalization(.never)
                           } else {
                               SecureField("sk-...", text: $apiKey)
                                   .textContentType(.password)
                           }
                           Button {
                               showAPIKey.toggle()
                           } label: {
                               Image(systemName: showAPIKey ? "eye.slash" : "eye")
                                   .foregroundStyle(showAPIKey ? Theme.accent : .secondary)
                           }
                           .buttonStyle(.plain)
                       }

                       Button("Save API Key") {
                           let trimmed = apiKey.trimmingCharacters(in: .whitespacesAndNewlines)
                           guard !trimmed.isEmpty else { return }
                           try? KeychainService.save(key: "openai_api_key", value: trimmed)
                           dismiss()
                           // Trigger retry after brief delay for dismiss animation
                           DispatchQueue.main.asyncAfter(deadline: .now() + 0.5) {
                               onSave()
                           }
                       }
                       .disabled(apiKey.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty)
                   } footer: {
                       Text("Required for AI transcription. Text-only scripts don't need a key.")
                   }
               }
               .navigationTitle("OpenAI API Key Required")
               .navigationBarTitleDisplayMode(.inline)
               .toolbar {
                   ToolbarItem(placement: .cancellationAction) {
                       Button("Cancel") { dismiss() }
                   }
               }
           }
       }
   }
   ```

   Key details per UI-SPEC Screen 6 and Copywriting Contract:
   - Sheet title: "OpenAI API Key Required"
   - Same SecureField + eye toggle pattern as Settings (D-14)
   - Footer: "Required for AI transcription. Text-only scripts don't need a key."
   - Save button dismisses sheet AND calls `onSave` which retries transcription

   - If `content.contentType == .text`: Show nothing (EmptyView)

   - If transcript is nil: Show TranscribeButtonView in `.idle` (or `.error` if last attempt failed)

   Replace the placeholder Transcribe Audio button action with:
   ```swift
   private func startTranscription() {
       Task {
           transcribeState = .inProgress
           do {
               let text = try await transcriptionService.transcribe(content: content)
               content.transcript = text  // TRANS-03: save to SwiftData model
               transcribeState = .complete
           } catch let error as TranscriptionError {
               switch error {
               case .noAPIKey:
                   transcribeState = .idle
                   showAPIKeyPrompt = true
               case .noNetwork:
                   transcribeState = .error("Transcription failed. Check your connection and try again.")
               case .fileTooLarge:
                   transcribeState = .error("Audio is too long to transcribe. Try a shorter clip.")
               default:
                   transcribeState = .error(error.localizedDescription)
               }
           } catch {
               transcribeState = .error(error.localizedDescription)
           }
       }
   }
   ```

   Add `.sheet(isPresented: $showAPIKeyPrompt)` presenting `APIKeyPromptSheet(onSave: startTranscription)` — per D-13: on save, transcription auto-retries.

   Set initial `transcribeState` in `.onAppear`:
   ```swift
   .onAppear {
       if content.transcript != nil {
           transcribeState = .complete
       }
   }
   ```

   The full updated `transcriptSection` should look like:
   ```swift
   @ViewBuilder
   private var transcriptSection: some View {
       if content.contentType != .text {
           TranscribeButtonView(state: transcribeState, onTranscribe: startTranscription)

           if let transcript = content.transcript {
               Text(transcript)
                   .font(.body)
                   .padding(.top, 8)
           }
       }
   }
   ```

   Add the API key prompt sheet modifier to the NavigationStack:
   ```swift
   .sheet(isPresented: $showAPIKeyPrompt) {
       APIKeyPromptSheet(onSave: startTranscription)
   }
   ```
